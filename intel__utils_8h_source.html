73 // Runs example function with signature void() and catches errors.
74 // Returns `0` on success, `1` or oneDNN error, and `2` on example error.
75 inline int handle_example_errors(
76  std::initializer_list<dnnl::engine::kind> engine_kinds,
77  std::function<void()> example) {
78  int exit_code = 0;
79 
80  try {
81  example();
82  } catch (example_allows_unimplemented &e) {
83  std::cout << e.message << std::endl;
84  exit_code = 0;
85  } catch (dnnl::error &e) {
86  std::cout << "oneDNN error caught: " << std::endl
87  << "\tStatus: " << dnnl_status2str(e.status) << std::endl
88  << "\tMessage: " << e.what() << std::endl;
89  exit_code = 1;
90  } catch (std::exception &e) {
91  std::cout << "Error in the example: " << e.what() << "." << std::endl;
92  exit_code = 2;
93  }
94 
95  std::string engine_kind_str;
96  for (auto it = engine_kinds.begin(); it != engine_kinds.end(); ++it) {
97  if (it != engine_kinds.begin()) engine_kind_str += "/";
98  engine_kind_str += engine_kind2str_upper(*it);
99  }
100 
101  std::cout << "Example " << (exit_code ? "failed" : "passed") << " on "
102  << engine_kind_str << "." << std::endl;
103  return exit_code;
104 }

123 inline dnnl::engine::kind parse_engine_kind(
124  int argc, char **argv, int extra_args) {
125  // Returns default engine kind, i.e. CPU, if none given
126  if (argc == 1) {
127  return validate_engine_kind(dnnl::engine::kind::cpu);
128  } else if (argc <= extra_args + 2) {
129  std::string engine_kind_str = argv[1];
130  // Checking the engine type, i.e. CPU or GPU
131  if (engine_kind_str == "cpu") {
132  return validate_engine_kind(dnnl::engine::kind::cpu);
133  } else if (engine_kind_str == "gpu") {
134  return validate_engine_kind(dnnl::engine::kind::gpu);
135  }
136  }
137 
138  // If all above fails, the example should be ran properly
139  std::cout << "Inappropriate engine kind." << std::endl
140  << "Please run the example like this: " << argv[0] << " [cpu|gpu]"
141  << (extra_args ? " [extra arguments]" : "") << "." << std::endl;
142  exit(1);
143 }

157 // Read from memory, write to handle
158 inline void read_from_dnnl_memory(void *handle, dnnl::memory &mem) {
159  dnnl::engine eng = mem.get_engine();
160  size_t size = mem.get_desc().get_size();
161 
162  if (!handle) throw std::runtime_error("handle is nullptr.");
163 
164 #ifdef DNNL_WITH_SYCL
165  bool is_cpu_sycl = (DNNL_CPU_RUNTIME == DNNL_RUNTIME_SYCL
166  && eng.get_kind() == dnnl::engine::kind::cpu);
167  bool is_gpu_sycl = (DNNL_GPU_RUNTIME == DNNL_RUNTIME_SYCL
168  && eng.get_kind() == dnnl::engine::kind::gpu);
169  if (is_cpu_sycl || is_gpu_sycl) {
170  auto mkind = dnnl::sycl_interop::get_memory_kind(mem);
171  if (mkind == dnnl::sycl_interop::memory_kind::buffer) {
172  auto buffer = dnnl::sycl_interop::get_buffer<uint8_t>(mem);
173  auto src = buffer.get_access<cl::sycl::access::mode::read>();
174  uint8_t *src_ptr = src.get_pointer();
175  if (!src_ptr)
176  throw std::runtime_error("get_pointer returned nullptr.");
177  for (size_t i = 0; i < size; ++i)
178  ((uint8_t *)handle)[i] = src_ptr[i];
179  } else {
180  assert(mkind == dnnl::sycl_interop::memory_kind::usm);
181  uint8_t *src_ptr = (uint8_t *)mem.get_data_handle();
182  if (!src_ptr)
183  throw std::runtime_error("get_data_handle returned nullptr.");
184  if (is_cpu_sycl) {
185  for (size_t i = 0; i < size; ++i)
186  ((uint8_t *)handle)[i] = src_ptr[i];
187  } else {
188  auto sycl_queue
189  = dnnl::sycl_interop::get_queue(dnnl::stream(eng));
190  sycl_queue.memcpy(handle, src_ptr, size).wait();
191  }
192  }
193  return;
194  }
195 #endif
196 #if DNNL_GPU_RUNTIME == DNNL_RUNTIME_OCL
197  if (eng.get_kind() == dnnl::engine::kind::gpu) {
198  dnnl::stream s(eng);
199  cl_command_queue q = dnnl::ocl_interop::get_command_queue(s);
200  cl_mem m = dnnl::ocl_interop::get_mem_object(mem);
201 
202  cl_int ret = clEnqueueReadBuffer(
203  q, m, CL_TRUE, 0, size, handle, 0, NULL, NULL);
204  if (ret != CL_SUCCESS)
205  throw std::runtime_error("clEnqueueReadBuffer failed.");
206  return;
207  }
208 #endif
209 
210  if (eng.get_kind() == dnnl::engine::kind::cpu) {
211  uint8_t *src = static_cast<uint8_t *>(mem.get_data_handle());
212  if (!src) throw std::runtime_error("get_data_handle returned nullptr.");
213  for (size_t i = 0; i < size; ++i)
214  ((uint8_t *)handle)[i] = src[i];
215  return;
216  }
217 
218  assert(!"not expected");
219 }

223  dnnl::engine eng = mem.get_engine();
224  size_t size = mem.get_desc().get_size();
225 
226  if (!handle) throw std::runtime_error("handle is nullptr.");
